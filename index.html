async function uploadToFile() { const fileInput = document.getElementById('fileInput'); const status = document.getElementById('status'); if (fileInput.files.length === 0) return alert("Dosya seçmedin!"); const file = fileInput.files[0]; const reader = new FileReader(); status.innerText = "Yükleniyor... ⏳"; reader.onload = async function() { const base64Content = reader.result.split(',')[1]; // BURAYA YENİ YAYINLADIĞIN GOOGLE SCRIPT URL'SİNİ YAZ const SCRIPT_URL = "https://script.google.com/macros/s/..../exec"; try { const response = await fetch(SCRIPT_URL, { method: "POST", body: JSON.stringify({ fileName: file.name, content: base64Content }) }); const result = await response.json(); if (result.status === "success") { // Linkini Vercel domainine göre ayarlar const siteUrl = window.location.origin + "/"; const finalLink = `${siteUrl}download.html?file=${result.path}&name=${result.fileName}`; status.innerHTML = `✅ Yüklendi!
${finalLink}
`; } else { status.innerText = "❌ Hata: " + result.msg; } } catch (e) { status.innerText = "⚠️ Bağlantı hatası!"; } }; reader.readAsDataURL(file); }
